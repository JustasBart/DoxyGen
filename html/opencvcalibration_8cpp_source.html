393 static double computeReprojectionErrors( const vector<vector<Point3f> >& objectPoints,
394  const vector<vector<Point2f> >& imagePoints,
395  const vector<Mat>& rvecs, const vector<Mat>& tvecs,
396  const Mat& cameraMatrix , const Mat& distCoeffs,
397  vector<float>& perViewErrors)
398 {
399  vector<Point2f> imagePoints2;
400  int i, totalPoints = 0;
401  double totalErr = 0, err;
402  perViewErrors.resize(objectPoints.size());
403 
404  for( i = 0; i < (int)objectPoints.size(); ++i )
405  {
406  projectPoints( Mat(objectPoints[i]), rvecs[i], tvecs[i], cameraMatrix,
407  distCoeffs, imagePoints2);
408  err = norm(Mat(imagePoints[i]), Mat(imagePoints2), CV_L2);
409 
410  int n = (int)objectPoints[i].size();
411  perViewErrors[i] = (float) std::sqrt(err*err/n);
412  totalErr += err*err;
413  totalPoints += n;
414  }
415 
416  return std::sqrt(totalErr/totalPoints);
417 }
